#Global Reserve Currency

global_reserve_currency = {
	trigger = {
		rank = 2
		big_producer = credit
		OR = { government = democracy government = hms_government government = absolute_monarchy }
	}
	prestige = 0.01
	import_cost = -0.40
	diplomatic_points_modifier = 0.50
	rich_income_modifier = 0.2
	middle_income_modifier = 0.1
	icon = 5
}

major_reserve_currency = {
	trigger = {
		NOT = { rank = 2 }
		rank = 8
		big_producer = credit
