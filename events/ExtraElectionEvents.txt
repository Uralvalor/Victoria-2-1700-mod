country_event = {
	id = 98300
	title = "EVTNAME98300"
	desc = "EVTDESC98300"
	picture = "election_crowd"
	
	trigger = {
		election = yes
		war = yes
		NOT = {
			has_country_flag = election_initialized
		}
	}
	
	mean_time_to_happen = {
		days = 15
		modifier = {
			factor = 0.5
			any_greater_power = { war_with = THIS }
		}
		modifier = {
			factor = 0.5
			any_neighbor_country = {
				OR = {
					is_secondary_power = yes
					is_greater_power = yes
				}
				war_with = THIS
			}
		}
		modifier = {
			factor = 0.5
			is_secondary_power = no
			is_greater_power = no
		}
		modifier = {
			factor = 0.5
			war_exhaustion = 20
		}
		modifier = {
			factor = 0.5
			war_exhaustion = 40
		}
		modifier = {
			factor = 1.5
			NOT = { war_exhaustion = 20 }
		}
		modifier = {
			factor = 0.1
			any_owned_province = {
				NOT = { controlled_by = THIS }
				NOT = { controlled_by = REB }
			}
		}
		modifier = {
			factor = 3
			is_greater_power = yes
			NOT = { any_greater_power = { war_with = THIS } }
			NOT = {
				any_owned_province = {
					NOT = { controlled_by = THIS }
					NOT = { controlled_by = REB }
				}
			}
		}
	}
	
	option = {
		name = "EVTOPTA98300"
		set_country_flag = election_initialized
		plurality = -1
		add_country_modifier = {
			name = wartime_elections
			duration = -1
		}
		add_country_modifier = {
			name = election_cooldown
			duration = 10 
		}
		random_owned = {
			limit = {
				owner = { is_ideology_enabled = fascist }
			}
			owner = {
				any_pop = {
					limit = {
						location = { is_colonial = no }
						is_primary_culture = yes
					}
					ideology = {
						value = fascist
						factor = 0.05
					}
				}
			}
		}
		random_owned = {
			limit = {
				owner = { NOT = { is_ideology_enabled = fascist } }
			}
			owner = {
				any_pop = {
					limit = {
						location = { is_colonial = no }
						OR = {
							type = soldiers
							type = officers
							type = aristocrats
						}
					}
					ideology = {
						value = reactionary
						factor = 0.05
					}
				}
			}
		}
		any_pop = {
			limit = {
				location = { is_core = THIS }
				is_primary_culture = yes
			}
			dominant_issue = {
				factor = 0.05
				value = jingoism
			}
		}
		any_pop = {
			limit = {
				location = { is_core = THIS }
				is_primary_culture = no
			}
			dominant_issue = {
				factor = 0.05
				value = pacifism
			}
		}
	}
}
