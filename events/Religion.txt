# Separation of Church and State
country_event = {
	id = 4432
	title = EVT_4432_NAME # Separation of Church and State
	desc = EVT_4432_DESC # From the first day of rule, it was obvious that the new $COUNTRY_ADJ$ government had a different attitude towards religion than the old government apparatus. It began with the attitudes of judges appointed to the courts and cuts in spending to religion institutions. With the latest instructions, the prevention of outward preference toward any religion in state institutions, it is clear that religion is no longer part of government.


	trigger = {
		NOT = { OR = { religious_policy = moralism religious_policy = pluralism } }
		has_country_flag = desecularized
		NOT = { government = theocracy }
	}

	mean_time_to_happen = {
		months = 12
		modifier = {
			factor = 0.4
			religious_policy = pro_atheism
		}
	}

	option = {
		name = EVT_4432_A # Real, visible progress
		any_pop = { limit = { is_state_religion = no } consciousness = -1 }
		any_pop = { scaled_consciousness = { factor = 1 issue = moralism } }
		religion = secularism
		clr_country_flag = desecularized
	}
}
